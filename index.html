#!/bin/sh set -ex SUDO="$(command -v sudo || true) " CONTAINER=$(grep -q docker /proc/self/cgroup && echo "true" || true) setup() { for PKG in "$@" do if grep -E " $PKG$" "$INSTALL_LOGS/installed" then echo "\033[94mSkipping $PKG - already installed\033[0m" else get "$INSTALLER/$PKG" "$SCRIPTS/$PKG" || get "common/$PKG" "$SCRIPTS/$PKG" if [ ! -f "$SCRIPTS/$PKG" ] then echo "❌ Package not available: ${PKG}" exit 1 fi . "$SCRIPTS/$PKG" 2>> "$INSTALL_LOGS/log" echo "$(date) $PKG" >> "$INSTALL_LOGS/installed" fi done } install() { # TODO: check if each pkg is available and build list of pending installs case $INSTALLER in apt) sudo apt install -y $@ ;; dnf) sudo dnf install -y $@ ;; pacman) sudo pacman -S --needed --noconfirm $@ ;; esac } get() { curl -sfS "https://raw.githubusercontent.com/Funk66/installer/master/$1" -o "$2" --create-dirs 2> /dev/null } available() { command -v "$1" > /dev/null 2>&1 } sudo() { ${SUDO}$@ } bwget() { [ -z "$BW_SESSION" ] && export BW_SESSION=$(bw unlock --raw) bw get item "$1" | jq -r ".notes" } for INSTALLER in apt dnf pacman do available $INSTALLER && break done if [ -z $INSTALLER ] then echo "❌ Unable to recognize OS" exit 1 fi SCRIPTS=$(mktemp -d) INSTALL_LOGS=~/.local/funk66 mkdir -p $INSTALL_LOGS touch $INSTALL_LOGS/installed rm "$INSTALL_LOGS/log" setup "$@" rm -r "$SCRIPTS" echo "✅ Done"
